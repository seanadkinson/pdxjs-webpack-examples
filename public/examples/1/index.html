Simple Webpack Bundle: Hello World

main.js

var greet = require('./greet');
document.getElementById('greet').onclick = function() {
    var who = document.getElementById('who').value;
    greet(who);
};

greet.js

function greet(who) {
    alert("Hello, " + who);
}
module.exports = greet;

Let's bundle this module together:

  • Webpack can easily be installed from npm 
        $ npm install -g webpack
  • Provide webpack a configuration file webpack.config.js 
    module.exports = {
    entry: './main.js',
    output: {
        filename: './bundle.js'
    }
};
  • Run webpack in the directory 
    $  webpack
